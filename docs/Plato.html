def carbo_total
    aux = 0
    @alimentos.each do |x|
        aux += x.carbo
    end
    @carb = aux.round(2)
    @p_carb = (((100 * carb) / @cantidades)).round(2)
end

def to_s
    os = ""
    os += "Nombre del plato: #{@nombre}\n" 
    @alimentos.each do |x|
        # os += x.to_s + "\n"
        os += x.nombre + " -> " + x.gramos.to_s + " gramos\n"
    end
    os += "Cantidad total: #{@cantidades}"
    os
end
