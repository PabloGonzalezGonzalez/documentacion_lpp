def impacto_huella
    if (@gei_t < 800)
        return 1

    elsif (@gei_t <= 1200)
        return 2

    else
        return 3
    end
end

def impacto_total
    i_ener = impacto_energia
    i_gei = impacto_huella

    return ( (i_ener + i_gei) / 2).round(1)
end
